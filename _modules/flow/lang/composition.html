def graph() -> Iterator[Any]:
    """
    A context manager that should be used to create flow graphs. Using this
    context manager creates a new :py:mod:`flow.model.graph.Model` and makes it
    active in the `with` context.
    """
    global current_model
    current = current_model
    try:
        current_model = Model()
        yield current_model
    finally:
        current_model = current



def node_count() -> int:
    if current_model is None:
        raise ValueError("node_count() is only valid in a graph() context")
    return len(current_model.nodes)


class Channel(IChannel):
    dest: Optional[IPort]

    def __init__(self, name: str, src: IPort) -> None:
        self.name = name
        self.src = src
        self.dest = None

    def connect(self, dest: IPort) -> IChannel:
        self.dest = dest
        if current_model is None:
            raise ValueError("No current model, cannot connect ports")

        current_model.add_channel(
            self.src.node.name,
            PortSpec(**asdict(self.src)),
            dest.node.name,
            PortSpec(**asdict(dest)),
        )
        return self


class Output(IOutput, IPipelinable):
    def is_multi_output(self) -> bool:
        return False

    def feed_in(self, i: IPipelinable) -> IPipelinable:
        # This should not be possible from the APIs: it's an internal error
        raise AssertionError("cannot feed into an output")  # pragma: no cover

    def feed_out(self) -> IConnectable:
        channel = Channel(f"{self.name}[{len(self.outputs)}]", self)
        self.outputs.append(channel)
        return channel

    def __rshift__(self, other: IPipelinable) -> IEndPoint:
        return Pipeline(self, other.feed_in(self))


class Input(IPort):
    input: Optional[IChannel]

    def __init__(self, node: "Node", spec: PortSpec) -> None:
        super().__init__(node, spec)
        self.input = None

    def is_multi_output(self) -> bool:
        return False

    def feed_in(self, i: IPipelinable) -> IPipelinable:
        if self.input is not None:
            raise ValueError(
                f"input {self.name} of {self.node.name} is already connected."
            )

        self.input = i.feed_out().connect(self)
        return self

    def feed_out(self) -> IConnectable:
        # This should not be possible from the APIs: it's an internal error
        raise AssertionError("cannot feed out of an input")  # pragma: no cover


class Bundle(IEndPoint, IConnectable, Mapping[str, IPipelinable]):
    def __init__(self, members: Mapping[str, IPipelinable]):
        self.members = set(members.keys())
        self.pipelines: Dict[str, IPipelinable] = dict(members)

    def is_multi_output(self) -> bool:
        return len(self) > 1

    def from_call(self, *args: IPipelinable, **kwargs: IPipelinable) -> IPipelinable:
        bad_keys = kwargs.keys() - self.pipelines.keys()
        if len(bad_keys) != 0:
            raise ValueError(f"no such pipeline in bundle: {', '.join(bad_keys)}")
        if len(self) < len(args):
            raise ValueError("too many arguments")
        for i, key in enumerate(self.pipelines):
            if i == len(args):
                break
            if key in kwargs:
                raise ValueError(
                    f"argument {key} passed as regular and keyword argument"
                )
            kwargs[key] = args[i]
        # raise error on args we've missed
        if len(kwargs) == 0:
            return self
        else:
            return self.feed_in(Bundle(kwargs))

    def feed_in(self, i: IPipelinable) -> IPipelinable:
        if len(self) == 1 and not i.is_multi_output():
            list(self.pipelines.values())[0].feed_in(i)
        else:
            bundle = i.feed_out()
            assert isinstance(bundle, Bundle), "expected a Bundle"
            overlap = self.members & bundle.members
            if len(overlap) == 0:
                raise CompositionError(
                    f"Don't know how to connect ({', '.join(self.members)}) with ({', '.join(bundle.members)})"
                )
            for member in overlap:
                self.pipelines[member].feed_in(bundle[member])
        return self

    def feed_out(self) -> IConnectable:
        return self

    def connect(self, dest: IPort) -> IChannel:
        if len(self) == 0:
            raise ValueError("no pipelines to connect")
        elif len(self) == 1:
            return list(self.pipelines.values())[0].feed_out().connect(dest)
        else:
            return self.pipelines[dest.name].feed_out().connect(dest)

    def __len__(self) -> int:
        return len(self.pipelines)

    def __iter__(self) -> Iterator[str]:
        return self.pipelines.__iter__()

    def __getitem__(self, member: str) -> IPipelinable:
        return self.pipelines[member]

    def __call__(self, *args: IPipelinable, **kwargs: IPipelinable) -> IEndPoint:
        self.from_call(*args, **kwargs)
        return self

    def __rshift__(self, other: IPipelinable) -> IEndPoint:
        return Pipeline(self, other.feed_in(self))


class Rebundle(IPipelinable):
    def __init__(self, **renames: str):
        revmap = {v: k for k, v in renames.items()}
        self.members = set(revmap.keys())
        self.renames = revmap

    def is_multi_output(self) -> bool:
        return len(self.members) > 1

    def feed_out(self) -> IConnectable:
        raise AssertionError("cannot feed out of a rebundleper")  # pragma: no cover

    def feed_in(self, i: IPipelinable) -> IPipelinable:
        bundle = i.feed_out()
        assert isinstance(bundle, Bundle), "expected a Bundle"
        overlap = self.members & bundle.members
        if len(overlap) == 0:
            raise CompositionError(
                f"Don't know how to connect ({', '.join(self.members)}) with ({', '.join(bundle.members)})"
            )
        pipelines: Dict[str, IPipelinable] = {}
        for member in bundle:
            if member in self.renames:
                pipelines[self.renames[member]] = bundle[member]
            else:
                pipelines[member] = bundle[member]
        return Bundle(pipelines)


class NodeView(IEndPoint, Mapping[str, IPort]):
    def __init__(
        self,
        inputs: Mapping[str, IPort],
        outputs: Mapping[str, IPort],
    ):
        self._inputs = dict(inputs)
        self._outputs = dict(outputs)

    def __call__(self, *args: IPipelinable, **kwargs: IPipelinable) -> IEndPoint:
        Bundle(self._inputs).from_call(*args, **kwargs)
        return self

    def is_multi_output(self) -> bool:
        return len(self._outputs) > 1

    def feed_in(self, i: IPipelinable) -> IPipelinable:
        Bundle(self._inputs).feed_in(i)
        return self

    def feed_out(self) -> IConnectable:
        return Bundle(self._outputs)

    def __rshift__(self, other: IPipelinable) -> IEndPoint:
        return Pipeline(self, other.feed_in(self))

    def __len__(self) -> int:
        return len(self._inputs) + len(self._outputs)

    def __iter__(self) -> Iterator[str]:
        return chain(self._inputs.__iter__(), self._outputs.__iter__())

    def _find_members(
        self, names: Iterable[str], *, none_is_all: bool = True
    ) -> Tuple[Sequence[str], Sequence[str]]:
        inmembers = [m for m in names if m in self._inputs]
        outmembers = [m for m in names if m in self._outputs]

        unknown = set(names) - (set(inmembers) | set(outmembers))
        if len(unknown) != 0:
            raise ValueError(f"no such port {', '.join(unknown)}")

        if none_is_all:
            if len(inmembers) == 0:
                inmembers = list(self._inputs)
            if len(outmembers) == 0:
                outmembers = list(self._outputs)
        return inmembers, outmembers

    @property
    def inputs(self) -> "NodeView":
        return NodeView(inputs=self._inputs, outputs={})

    @property
    def outputs(self) -> "NodeView":
        return NodeView(inputs={}, outputs=self._outputs)

    @overload
    def __getitem__(self, members: str) -> IPort:
        ...

    @overload
    def __getitem__(self, members: Tuple[str]) -> "NodeView":
        ...

    def __getitem__(self, members: Union[str, Tuple[str]]) -> Union["NodeView", IPort]:
        if isinstance(members, str):
            if members in self._inputs and members in self._outputs:
                raise ValueError(f"multiple ports called '{members}'")
            result = self._inputs.get(members, self._outputs.get(members, None))
            if result is None:
                raise ValueError(f"no such port {members}")
            return result

        inmembers, outmembers = self._find_members(members)

        ins = {m: self._inputs[m] for m in inmembers}
        outs = {m: self._outputs[m] for m in outmembers}
        return NodeView(ins, outs)

    def _pos2key(self, *args: Any) -> Mapping[str, Any]:
        allports = list(self._inputs.keys())
        allports.extend(list(self._outputs.keys()))
        return {k: args[i] for i, k in enumerate(allports[: len(args)])}

    def view(self, *ports: str, **renames: str) -> "NodeView":
        selection = {**renames, **{p: p for p in ports}}
        members = selection.keys()
        inmembers, outmembers = self._find_members(members)

        ins = {selection.get(m, m): self._inputs[m] for m in inmembers}
        outs = {selection.get(m, m): self._outputs[m] for m in outmembers}
        return NodeView(ins, outs)


class Node(INode, NodeView):
    def __init__(
        self,
        name: str,
        runner: Any = None,
        inputs: Sequence[PortSpec] = [],
        outputs: Sequence[PortSpec] = [],
    ):
        if current_model is None:
            raise ValueError("No current model, cannot construct a node")
        if name in current_model:
            raise ValueError(f"Node '{name}' already exists")
        inset = {i.name for i in inputs}
        outset = {o.name for o in outputs}
        if len(inset) != len(inputs) or len(outset) != len(outputs):
            raise ValueError(f"Node '{name}' has duplicate port names")

        self.name = name
        self.runner = runner
        super().__init__(
            {i.name: Input(self, i) for i in inputs},
            {o.name: Output(self, o) for o in outputs},
        )
        logger.info(
            f"created node {self}({', '.join(i.name for i in inputs)})"
            " -> ({', '.join(o.name for o in outputs)})"
        )
        current_model.add_actor(self.name, self.runner)



[docs]class Pipeline(IEndPoint):
    "A helper class that allows nodes to be connected with >>"

    def __init__(self, head: IPipelinable, tail: IPipelinable) -> None:
        self._head = head
        self._tail = tail

    def is_multi_output(self) -> bool:
        return self._tail.is_multi_output()

    def feed_in(self, i: IPipelinable) -> IPipelinable:
        self._head.feed_in(i)
        return self

    def feed_out(self) -> IConnectable:
        return self._tail.feed_out()

    def __rshift__(self, other: Any) -> IEndPoint:
        return Pipeline(self._head, other.feed_in(self._tail))

    def __call__(self, *args: IPipelinable, **kwargs: IPipelinable) -> IEndPoint:
        if not isinstance(self._head, IEndPoint):
            raise ValueError(
                "Head of Pipeline is not an endpoint: cannot connect inputs"
            )
        self._head(*args, **kwargs)
        return self
