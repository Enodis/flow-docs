"""
Execute the nodes of a data flow graph with dynamic event-driven system.

Async execution of a data flow graph is useful when data is received, live,
from external sources. In streaming execution the node cannot get the entire
input data as the data may be continuous, so also the output data must be
written piecemeal.
"""
import asyncio
import logging
from abc import ABC, abstractmethod
from typing import (
    Any,
    Dict,
    Iterable,
    List,
    Mapping,
    Sequence,
    Set,
    TYPE_CHECKING,
    Union,
)
from warnings import warn

from flow.model import IteratorRunner, Model, StatefulRunner, StatelessRunner, PortSpec
from flow.processor.asynch.channelqueue import (
    QueueClosedError,
)
from flow.processor.asynch.channelio import ChannelIO, ReaderFront, WriterFront

if TYPE_CHECKING:  # pragma: no cover
    Task = asyncio.Task[Any]
else:
    Task = asyncio.Task

logger = logging.getLogger(__name__)


State = Dict[str, ChannelIO[Any]]
EdgeMap = Mapping[str, PortSpec]


def _reraise_exception(tasks: Iterable[Task]) -> None:
    raiser = None
    # Fetch all exceptions or we get an error
    for task in tasks:
        exc = task.exception()
        if raiser is None or isinstance(raiser, QueueClosedError):
            raiser = exc

    if exc is not None:
        raise exc
    return None

[docs]class AsyncOperation(IAsyncOperation):
    """
    Common base for streaming operations with the sequence: read inputs,
    execute, write outputs.
    """

    _inmap: EdgeMap  #: Mapping from queue names to input names
    _outmap: EdgeMap  #: Mapping from queue names to output names

    #: Mapping from input names to the `ReaderFront` of queues
    _inq: Mapping[str, ReaderFront[Any]]

    #: Mapping from output names to the `WriterFront` of queues
    _outq: Mapping[str, WriterFront[Any]]

    def __init__(
        self,
        name: str,
        inmap: EdgeMap,
        outmap: EdgeMap,
    ) -> None:
        self.name = name
        self._inmap = inmap
        self._outmap = outmap
        self._inq = {}
        self._outq = {}


[docs]    def initialize(self, state: State) -> None:
        self._inq = {q: state[q].reader for q, port in self._inmap.items()}
        self._outq = {q: state[q].writer for q, port in self._outmap.items()}



[docs]    async def terminate(self) -> None:
        """
        Close all input and output queues.

        This method should not be called directly, but it may be overriden in
        subclasses. Termination of a node should be done by returning from
        `AsyncOperation.main`, normally or with an exception.
        """
        for wq in self._outq.values():
            await wq.close()
        for rq in self._inq.values():
            await rq.close()



[docs]    async def execute(self) -> None:
        try:
            await self.main()
        except QueueClosedError:
            pass
        finally:
            await self.terminate()



[docs]    async def read_inputs(self) -> Dict[str, Any]:
        """
        Read values from the input queues, waiting when necessary.

        This method may raise `QueueClosedError`, the default behavior is to
        terminate the node, closing all connected queues.

        Returns:
            A dict with values for each input port.
        """
        assert len(self._inq) != 0

        # Wait for all inputs
        aws = {
            qname: asyncio.create_task(q.get(self._inmap[qname].make_container()))
            for qname, q in self._inq.items()
        }
        done, pending = await asyncio.wait(
            aws.values(), return_when=asyncio.FIRST_EXCEPTION
        )

        # Cancel remaining input tasks, only happens on exceptions
        for task in pending:
            task.cancel()

        _reraise_exception(done)

        # No failures: read all inputs
        return {self._inmap[q].name: task.result().unbox() for q, task in aws.items()}



[docs]    async def write_outputs(self, values: Mapping[str, Any]) -> None:
        """
        Write values to the output queues.

        Args:
            values: a dict with a value to put to each output port.
        """
        for var, port in self._outmap.items():
            q = self._outq[var]
            value = values[port.name]
            seq = isinstance(value, Sequence)

            logger.info(f"write {value} -> {self.name}.{port.name}")
            if port.window_size == 1 or not seq:
                if port.window_size != 1:
                    warn(f"expected a sequence output, got {type(value)}")
                # Note: it is NOT an error if `value` is a sequence of length>1 for
                # port.window_size==1, it just means that the single value is a
                # sequence.
                value = [value]
            else:
                if len(value) != port.window_size:
                    warn(
                        f"Unexpected window size returned: got {len(value)} expected {port.window_size}"
                    )
            await q.put(value)

[docs]    async def main(self) -> None:
        self.runner.initialize()
        while True:
            await self._execute1()



[docs]    async def terminate(self) -> None:
        self.runner.terminate()
        return await super().terminate()


    async def _execute1(self) -> None:
        values = await self.read_inputs()
        rv = self.runner.execute(values)
        await self.write_outputs(rv)

[docs]class AsyncProgram:
    """
    A program is the representation of a data flow graph that can be executed by
    a `AsyncProcessor`. No run-time entities are created yet, that will be
    done during execution of the program.

    The normal way of creating a AsyncProgram is by calling the
    `AsyncProgram.compile` class method.

    A streaming program specifies a collection of tasks to be executed
    simultaneously, and a collection of queues that represent the communication
    between the tasks.

    Args:
        tasks: the tasks that represent the actor nodes in the data flow graph.
        queues: the communication between the actor nodes in the data flow
        graph.
    """

    def __init__(
        self,
        tasks: Sequence[IAsyncOperation],
        queues: Mapping[str, Mapping[str, Any]],
    ):
        self.tasks = tasks
        self.queues = queues


[docs]    @classmethod
    def compile(cls, g: Model) -> "AsyncProgram":
        """
        Compile a data flow graph into a streaming program.

        Args:
            g: the data flow graph to compile.

        Returns:
            A program that contains the specification for the execution of the
            data flow graph for a `AsyncProcessor`.
        """
        g.configure()
        queues: Dict[str, Dict[str, Any]] = {}
        for src, dst, data in g.edges(data=True):
            name = f"{src}.{data['outport'].name}-{dst}.{data['inport'].name}"
            queues[name] = {"maxsize": data.get("queueSize", 0)}
            data["queue"] = name

        tasks: List[IAsyncOperation] = []
        for n in g:
            runner = g.nodes[n]["runner"]
            if isinstance(runner, StatelessRunner) or isinstance(
                runner, StatefulRunner
            ):
                inmap = {
                    data["queue"]: data["inport"]
                    for _, _, data in g.in_edges(n, data=True)
                }
                outmap = {
                    data["queue"]: data["outport"]
                    for _, _, data in g.out_edges(n, data=True)
                }
                if len(inmap) == 0:
                    raise ValueError(
                        "stateless runners require at least 1 input", runner
                    )

                tasks.append(AsyncTask(n, runner, inmap, outmap))
            elif isinstance(runner, IteratorRunner):
                outmap = {
                    data["queue"]: data["outport"]
                    for _, _, data in g.out_edges(n, data=True)
                }
                tasks.append(AsyncGenerator(n, runner, outmap))
            else:
                raise ValueError(f"Unsupported runner: {runner}")
        return cls(tasks, queues)

[docs]    async def execute(self, program: Union[Model, AsyncProgram]) -> None:
        """
        Execute a data flow graph.

        Args:
            program: a `Model` which will be compiled and executed, or a
            `AsyncProgram` which will be executed.
        """
        if isinstance(program, Model):
            program = AsyncProgram.compile(program)

        self.tasks: Set[Task] = set()
        self.queues: State = {
            name: ChannelIO(maxsize=q["maxsize"]) for name, q in program.queues.items()
        }
        for t in program.tasks:
            t.initialize(self.queues)
            task: Task = asyncio.create_task(t.execute(), name=t.name)
            self.tasks.add(task)
            task.add_done_callback(self.tasks.discard)

        await asyncio.gather(*self.tasks)
