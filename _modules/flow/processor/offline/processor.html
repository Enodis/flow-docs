"""
Execute the nodes of a data flow graph with a pre-compiled sequence of operations.

Offline execution of a data flow graph is possible when all input data for the
data flow graph is present before execution. In offline execution the node may
accept the entire input data, and produce all output data at once.
"""

import logging
from abc import ABC, abstractmethod
from typing import Any, Dict, Iterable, Iterator, List, Mapping, Sequence, Union

import networkx as nx
from flow.model import IteratorRunner, Model, PortSpec, StatefulRunner, StatelessRunner

from .channelio import ChannelIO

logger = logging.getLogger(__name__)

State = Dict[str, ChannelIO[Any]]
EdgeMap = Mapping[str, PortSpec]

[docs]class OfflineOperation(IOfflineOperation):
    """
    Common structure for an offline operation with separate steps for reading
    inputs, execution, and writing outputs.

    This class should not be used directly.
    """

    _inmap: EdgeMap  #: Mapping from input port name to state key
    _outmap: EdgeMap  #: Mapping from output port name to state key

    def __init__(
        self,
        name: str,
        inmap: EdgeMap,
        outmap: EdgeMap,
    ) -> None:
        self.name = name
        self._inmap = inmap
        self._outmap = outmap

    def __str__(self) -> str:
        ins = ", ".join(f"{k}:{v}" for k, v in self._inmap.items())
        outs = ", ".join(f"{k}:{v}" for k, v in self._outmap.items())
        return f"function {self.name} {ins} -> {outs}"


[docs]    def ready_to_read(self, state: State) -> bool:
        """
        Check whether there is sufficient input to read all required input
        values.

        Args:
            state: the processor state from which to retrieve the input values.
        """
        return all(
            port.window_size <= len(state[var].reader)
            for var, port in self._inmap.items()
        )



[docs]    def read_inputs(self, state: State) -> Dict[str, Any]:
        """
        Read all inputs from processor state.

        Args:
            state: the processor state from which to retrieve the input values.

        Returns:
            A dict with the value for each input name.
        """
        return {
            port.name: state[var].reader.get(port.make_container()).unbox()
            for var, port in self._inmap.items()
        }



[docs]    def write_outputs(self, state: State, values: Mapping[str, Any]) -> None:
        """
        Write output values to processor state.

        Args:
            state: the processor state to write the output values to
            values: a dict with a value for each output name
        """
        for var, port in self._outmap.items():
            logger.info(
                f"write ${var} <- {values[port.name]}{{{self.name}.{port.name}}}"
            )
            if var not in state:
                state[var] = ChannelIO()
            vs = values[port.name]
            c = port.make_container()
            state[var].writer.put(c.box(vs))




[docs]class OfflineFunction(OfflineOperation):
    """
    An operation that performs a function on inputs producing outputs.
    """

    def __init__(
        self,
        name: str,
        runner: Union[StatelessRunner, StatefulRunner],
        inmap: EdgeMap,
        outmap: EdgeMap,
    ) -> None:
        super().__init__(name, inmap, outmap)
        self.runner = runner


[docs]    def execute(self, state: State) -> None:
        self.runner.initialize()
        try:
            while self.ready_to_read(state):
                values = self.read_inputs(state)
                logger.info(f"execute {self.name}({values})")
                rv = self.runner.execute(values)
                self.write_outputs(state, rv)
        finally:
            self.runner.terminate()

        logger.info(f"execute {self.name}()")
        for rv in self.runner.execute():
            self.write_outputs(state, rv)




[docs]class OfflineProgram(Iterable[IOfflineOperation]):
    """
    A program is the representation of a data flow graph that can be executed by
    an `OfflineProcessor`. No run-time entities are created yet, that will be
    done during execution of the program.

    The normal way of creating an OfflineProgram is by calling the
    `OfflineProgram.compile` class method.

    An OfflineProgram is simply a sequence of operations that adhere to
    `IOfflineOperation`.

    Args:
        operations: the sequence of operations that comprise this program.
    """

    def __init__(self, operations: Sequence[IOfflineOperation]):
        self.operations = operations

    def __iter__(self) -> Iterator[IOfflineOperation]:
        for op in self.operations:
            yield (op)

    def __str__(self) -> str:
        return "\n".join(str(o) for o in self.operations)


[docs]    @classmethod
    def compile(cls, g: Model) -> "OfflineProgram":
        """
        Compile a data flow graph into an offline program.

        Args:
            g: the data flow graph to compile.

        Returns:
            A program that contains the specification for the execution of the
            data flow graph for an `OfflineProcessor`.
        """
        g.configure()
        for src, dst, data in g.edges(data=True):
            data["index"] = f"{src}.{data['outport'].name}-{dst}.{data['inport'].name}"

        operations: List[IOfflineOperation] = []
        for n in nx.topological_sort(g):
            runner = g.nodes[n]["runner"]
            if isinstance(runner, StatelessRunner) or isinstance(
                runner, StatefulRunner
            ):
                inmap = {
                    data["index"]: data["inport"]
                    for _, _, data in g.in_edges(n, data=True)
                }
                outmap = {
                    data["index"]: data["outport"]
                    for _, _, data in g.out_edges(n, data=True)
                }

                operations.append(OfflineFunction(n, runner, inmap, outmap))
            elif isinstance(runner, IteratorRunner):
                outmap = {
                    data["index"]: data["outport"]
                    for _, _, data in g.out_edges(n, data=True)
                }
                operations.append(OfflineIterator(n, runner, outmap))
            else:
                raise ValueError(f"Unsupported runner: {runner}")  # pragma: no cover
        return cls(operations)

[docs]    def execute(self, program: Union[OfflineProgram, Model]) -> None:
        """
        Execute a data flow graph.

        Args:
            program: a `Model` which will be compiled and executed, or an
                `OfflineProgram` which will be executed.
        """
        if isinstance(program, Model):
            program = OfflineProgram.compile(program)

        self.state: State = {}
        for op in program:
            op.execute(self.state)
